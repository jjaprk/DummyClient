def cmd_btn_test_prize():
    build_effect_scenario()


def build_prize_scenario( target_prize ):
    prize_scenario = dict()
    prize_scenario['active'] = True
    prize_scenario['count'] = 0
    prize_scenario['offset'] = 0
    prize_scenario['total_prize'] = target_prize
    prize_scenario['remain_prize'] = target_prize

    remain_prize = target_prize

    usable_prize_table = [0, 5000, 0, 10000, 0, 20000]
    index = 0

    while remain_prize > 0:
        prize = random.choice(usable_prize_table)

        if prize != 0 and prize <= remain_prize:
            prize_scenario[str(index)] = prize
            remain_prize -= prize

        index += 1

    index += random.randrange(3,10)
    prize_scenario[str(index)] = 'day'
    prize_scenario['count'] = index

    for key, val in prize_scenario.items():
        put_scenario_log( key + ' : ' + str(val) )


def build_effect_scenario():
    cmd_btn_clear_scenario_log()

    target_prize = 80000
    effect_scenario = dict()
    effect_scenario['active'] = True
    effect_scenario['count'] = 0
    effect_scenario['offset'] = 0
    effect_scenario['prize'] = target_prize

    if target_prize >= 500000:
        target_step = 5
    elif target_prize >= 100000:
        target_step = 4
    elif target_prize >= 10000:
        target_step = 3
    elif target_prize > 0:
        target_step = 2
    else:
        target_step = 1

    current_step = 0

    steps_string_list = ['night', 'turtle', 'jellyfish', 'shark', 'whale']
    need_space_dict = {'night': [10, 30], 'turtle': [20, 40], 'jellyfish': [40, 80], 'shark': [15, 40], 'whale': [50, 100]}

    offset = 0

    while current_step < target_step:
        step_string = steps_string_list[current_step]

        effect_scenario[str(offset)] = step_string

        space = random.randrange( need_space_dict[step_string][0], need_space_dict[step_string][1])
        offset += space

        current_step += 1

    effect_scenario['count'] = offset

    for key, val in effect_scenario.items():
        put_scenario_log( key + ' : ' + str(val) )

    build_prize_scenario( target_prize)
